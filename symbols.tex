% Nomenclature does automatically sort your symbols. However, I personally
% wanted a different order, so I used square brackets to add numbers that
% sorted them automatically like old code where you write on every tenth
% line so you've got room to add stuff in between! You can see examples
% of autosorting where numbers are the same, and why one might want more
% control.

% Also note use of custom-defined symbols from custom_defs.tex

% Roman symbols [00100-00900] - all have the same number here and it works
\nomenclature[00300]{$\rain$}{Rain (as a model hydrometeor species).}
\nomenclature[00300]{$\graupel$}{Graupel (as a model hydrometeor species).}
\nomenclature[00300]{$\ice$}{Ice crystals (as a model hydrometeor species). Note this refers to the species only, and not to the total frozer water content, which would include snow and graupel.}
\nomenclature[00300]{$\water$}{Cloud water (as a model hydrometeor species). Note this refers to the species only, and not to the total liquid water content, which would include rain.}
\nomenclature[00300]{$\snow$}{Snow (as a model hydrometeor species).}
\nomenclature[00300]{Z}{Universal coordinated time.}

% Italic symbols [01000-09000] - note the overlines come first in the two
% labelled 1000 as it reads the \ for the alphabetical order - not ideal.
\nomenclature[01000]{$a, b, \ldots $}{Arbitrary constants used in parametrisations. Values are defined at source with respect to their parametrisations.}
\nomenclature[01000]{$\overline{a}, \overline{b}, \ldots $}{Arbitrary constants used in parametrisations of average values. Values are defined at source with respect to their parametrisations.}
\nomenclature[01110]{$A_{\text{d}}$}{Surface area of filter in contact with a droplet in a droplet-freezing assay.}
\nomenclature[01110]{$A_{\text{f}}$}{Surface area of filter exposed to aerosol in a droplet-freezing assay.}
\nomenclature[03500]{$\inpconc$}{Number concentration of ice-nucleating particles.}
\nomenclature[03530]{$\inpimmersion$}{Number concentration of immersion-mode ice-nucleating particles.}
\nomenclature[03530]{$\inpcontact$}{Number concentration of contact-mode ice-nucleating particles.}
\nomenclature[08200]{$x$}{Placeholder variable for an unspecified CASIM hydrometeor or aerosol tracer.}
\nomenclature[08500]{$y$}{Placeholder variable for an unspecified CASIM hydrometeor or aerosol tracer.}
% Scripts [09000-10000]
\nomenclature[09000]{$\mathcal{N}_{x}$}{Particle size distribution of hydrometeor species $x$.}
% Greek symbols [>10000]
\nomenclature[29900]{$\eta_{\mathrm{c}}$}{Efficiency of contact freezing (model parameter).}
\nomenclature[30000]{$\eta_{\mathrm{i}}$}{Efficiency of immersion freezing (model parameter).}
